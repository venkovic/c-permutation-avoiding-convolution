\setcounter{algorithm}{5}
\begin{algorithm}[H]
\caption{Butterfly kernel of radix-4}
\label{alg:butterfly-kernel-radix-4}
\begin{algorithmic}[1]
\REQUIRE{$x\in\mathbb{C}^n$, $n=4^t$}
\ENSURE{$x:=A_{4,n}x$}
\FOR{$q=1,\dots,t$}
\STATE{$k:=4^q$}
\STATE{$n_b:=n/k$}
\STATE{$\ell:=k/4$}
\FOR{$b=1,\dots,n_b$}
\FOR{$j=1\dots,\ell$}
\STATE{$z_1:=x_{(b-1)k+j}$}
\STATE{$z_2:=\omega_k^{j-1}x_{(b-1)k+\ell+j}$}
\STATE{$z_3:=\omega_k^{2(j-1)}x_{(b-1)k+2\ell+j}$}
\STATE{$z_4:=\omega_k^{3(j-1)}x_{(b-1)k+3\ell+j}$}
\STATE{$\tau_1:=z_1+z_3$, $\tau_2:=z_1-z_3$}
\STATE{$\tau_3:=z_2+z_4$, $\tau_4:=z_2-z_4$}
\STATE{$x_{(b-1)k+j}:=\tau_1+\tau_3$}
\STATE{$x_{(b-1)k+\ell+j}:=\tau_2-i\cdot \tau_4$}
\STATE{$x_{(b-1)k+2\ell+j}:=\tau_1-\tau_3$}
\STATE{$x_{(b-1)k+3\ell+j}:=\tau_2+i\cdot \tau_4$}
\ENDFOR
\ENDFOR
\ENDFOR
\RETURN{$x$}
\end{algorithmic}
\end{algorithm}
